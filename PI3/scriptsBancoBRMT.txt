create table USUARIO
(
	ID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY(Start with 1, Increment by 1),
	USUARIO VARCHAR(500) not null,
	SENHA VARCHAR(500) not null
)

create table CLIENTE
(
	ID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY(Start with 1, Increment by 1),
	NOME VARCHAR(140) not null,
	ENDERECO VARCHAR(200) not null,
	EMAIL VARCHAR(200) not null,
	DATA_NASCIMENTO DATE not null,
	SEXO VARCHAR(45) not null,
	TELEFONE VARCHAR(35) not null,
	CPF VARCHAR(35) not null,
	CNPJ VARCHAR(35) not null,
	TIPO VARCHAR(35) not null,
	STATUS VARCHAR(45 not null
)

create table CATEGORIA_PRODUTO
(
	ID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY(Start with 1, Increment by 1),
	NOME VARCHAR(140) not null

create table PRODUTO
(
	ID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY(Start with 1, Increment by 1),
	ID_LOJA INTEGER not null,
	NOME VARCHAR(140) not null,
	QTD_PRODUTO INTEGER not null,
	VALOR DOUBLE not null,
	ID_CATEGORIA_PRODUTO INTEGER not null,
	COR VARCHAR(140),
	MARCA VARCHAR(140),
	STATUS VARCHAR(45)
)

create table VENDA
(
	ID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY(Start with 1, Increment by 1),
	NOME_VENDEDOR VARCHAR(140) not null,
	ID_CLIENTE INTEGER not null,
	ID_LOJA INTEGER not null,
	ID_PRODUTO INTEGER not null,
	ID_CATEGORIA_PRODUTO INTEGER not null,
	VALOR_DESCONTO DOUBLE not null,
	VALOR_FINAL DOUBLE not null,
	ID_FORMA_PGTO INTEGER not null,
	DATA_VENDA DATE NOT NULL WITH DEFAULT CURRENT DATE
